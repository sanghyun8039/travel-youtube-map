export interface PlaceResult {
  googlePlaceId: string
  name: string
  lat: number
  lng: number
}

export async function getGooglePlaceId(query: string): Promise<PlaceResult | null> {
  const apiKey = process.env.GOOGLE_MAPS_API_KEY
  if (!apiKey) return null

  const res = await fetch('https://places.googleapis.com/v1/places:searchText', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'X-Goog-Api-Key': apiKey,
      'X-Goog-FieldMask': 'places.id,places.displayName,places.location',
    },
    body: JSON.stringify({ textQuery: query }),
  })

  if (!res.ok) return null

  const data = await res.json()
  const place = data.places?.[0]
  if (!place) return null

  return {
    googlePlaceId: place.id as string,
    name: (place.displayName?.text ?? query) as string,
    lat: place.location.latitude as number,
    lng: place.location.longitude as number,
  }
}
```

export interface ChannelInfo {
  channelId: string   // URL 마지막 세그먼트 (@handle 또는 UCxxxx)
  channelName: string
  channelUrl: string
}

export async function fetchChannelInfo(videoId: string): Promise<ChannelInfo> {
  const res = await fetch(
    `https://www.youtube.com/oembed?url=https://www.youtube.com/watch?v=${videoId}&format=json`
  )
  if (!res.ok) throw new Error(`oEmbed failed: ${res.status}`)
  const data = await res.json()
  const channelUrl = data.author_url as string
  const parts = channelUrl.split('/')
  const channelId = parts[parts.length - 1]
  return {
    channelId,
    channelName: data.author_name as string,
    channelUrl,
  }
}
```

## Task 5: POST /api/save 라우트

**Files:**
- Create: `app/api/save/route.ts`

- [ ] **Step 1: `app/api/save/route.ts` 작성**

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { db } from '@/lib/db'
import { fetchChannelInfo } from '@/lib/youtube'
import { getGooglePlaceId } from '@/lib/places'
import type { AnalysisResult } from '@/lib/types'

export async function POST(req: NextRequest) {
  let body: AnalysisResult
  try {
    body = await req.json()
  } catch {
    return NextResponse.json({ success: false, error: 'Invalid JSON' }, { status: 400 })
  }

  if (!body.videoId || !body.videoTitle || !Array.isArray(body.items)) {
    return NextResponse.json({ success: false, error: 'Missing required fields' }, { status: 400 })
  }

  try {
    // 1. 채널 정보 취득
    const channelInfo = await fetchChannelInfo(body.videoId)

    // 2. Channel upsert
    const channel = await db.channel.upsert({
      where: { channelId: channelInfo.channelId },
      update: { channelName: channelInfo.channelName, channelUrl: channelInfo.channelUrl },
      create: {
        channelId: channelInfo.channelId,
        channelName: channelInfo.channelName,
        channelUrl: channelInfo.channelUrl,
      },
    })

    // 3. Video upsert
    const video = await db.video.upsert({
      where: { videoId: body.videoId },
      update: { title: body.videoTitle },
      create: {
        videoId: body.videoId,
        title: body.videoTitle,
        channelId: channel.id,
      },
    })

    // 4. 기존 VideoPlace 전체 삭제 (삭제 후 재삽입으로 편집 상태 정확 반영)
    await db.videoPlace.deleteMany({ where: { videoId: video.id } })

    // 5. 현재 items 기준으로 재삽입
    for (let i = 0; i < body.items.length; i++) {
      const item = body.items[i]
      let placeDbId: string | null = null

      if (item.hasCoords) {
        const placeResult = await getGooglePlaceId(`${item.place} ${item.city}`)
        if (placeResult) {
          const place = await db.place.upsert({
            where: { googlePlaceId: placeResult.googlePlaceId },
            update: {},
            create: {
              googlePlaceId: placeResult.googlePlaceId,
              name: placeResult.name,
              address: item.address ?? null,
              city: item.city,
              country: item.country,
              countryCode: item.countryCode,
              lat: placeResult.lat,
              lng: placeResult.lng,
            },
          })
          placeDbId = place.id
        }
      }

      await db.videoPlace.create({
        data: {
          videoId: video.id,
          placeId: placeDbId,
          timestamp: item.timestamp,
          localName: item.place,
          description: item.description,
          orderIndex: i,
        },
      })
    }

    return NextResponse.json({ success: true, savedCount: body.items.length })
  } catch (err) {
    const message = err instanceof Error ? err.message : 'Unknown error'
    return NextResponse.json({ success: false, error: message }, { status: 500 })
  }
}
```
